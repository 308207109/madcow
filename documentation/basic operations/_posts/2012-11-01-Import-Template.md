## Usage
{% highlight bash %}
import = "<<The name of the template file to run>>"
importTemplate = "<<The name of the template file to run>>"
{% endhighlight %}

## Examples
{% highlight bash %}
import = "MyTemplateFile"
importTemplate = "MyTemplateFile"
{% endhighlight %}
